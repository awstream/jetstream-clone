829 WindowLenFilter::process (OperatorChain * chain,
830  std::vector<boost::shared_ptr<Tuple> > & tuples,
831  DataplaneMessage& window_marker) {
832  boost::lock_guard<boost::mutex> lock (boost::mutex);
833 
834  if ( k_in_win + tuples.size() > bound ) { //won't pass all tuples
835  unsigned tuples_to_pass = bound - k_in_win;
836 
837  if ( (err_field > -1) && (err_bound_lev == 0) ) {
838  err_bound_lev = jetstream::numeric(*tuples[tuples_to_pass], err_field);
839  LOG(INFO) << "------- Local cutoff: "<<err_bound_lev << "-------";
840  }
841 
842  tuples.resize(tuples_to_pass);
843  }
844  k_in_win += tuples.size();
845 
846 
847 
848  if ( window_marker.type() == DataplaneMessage::END_OF_WINDOW) {
849  boost::lock_guard<boost::mutex> lock (boost::mutex);
850  vector<double> ratios;
851  vector<unsigned> bounds;
852  unsigned cur_level = LEVELS-1;
853 // double fract_of_max = double(k_in_win) / bound;
854 // if (fract_of_theoretical < )
855 
856  for( int i = 0; i < LEVELS; ++i) {
857  unsigned bnd = k_in_win * double(i+1) / LEVELS;
858  bounds.push_back( bnd );
859  ratios.push_back( double(i+1) / LEVELS );
860  if ( (cur_level == LEVELS-1) && bnd >= bound)
861  cur_level = i;
862  }
863  bounds[LEVELS-1] = UINT_MAX;
864 
865  unsigned delta = congest_policy->get_step(id(), ratios.data(), ratios.size(), cur_level);
866 
867  bound = bounds[cur_level + delta];
868 
869  LOG_IF(INFO,delta != 0) << "Changing local thresh. New thresh is " << bound
870  << " and last-window had " << k_in_win;
871  LOG(INFO) << "thresholding error is " << err_bound_lev;
872  window_marker.set_tput_r2_threshold(err_bound_lev);
873  k_in_win = 0;
874  err_bound_lev = 0;
