80 inline void add_one_el(std::ostringstream& buf, const Element& el) {
81  bool had_el = false;

95  if (el.has_t_val()) {

102  buf << tmbuf<< " ";
103  had_el = true;
104  } else if (el.has_summary()) {
105  if(el.summary().has_histo()) {
106  const JSHistogram & h = el.summary().histo();
107  unsigned tally = 0;
108  for (size_t i = 0; i < (size_t) h.bucket_vals_size(); ++i)
109  tally += h.bucket_vals(i);
110  buf << "histogram(" << h.bucket_vals_size()<<")-" << tally << " elems";
111  } else {
112  buf << "misc. summary";
113  }
114  } else if (el.has_blob()) {
115  buf << el.blob().size() << "-byte blob";
116  } else
117  if(!had_el)
118  buf << "UNDEF";
119 }
