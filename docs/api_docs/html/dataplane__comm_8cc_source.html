639

641  return;
642 
643  int active_chains = 0;
644  for (unsigned i = 0; i < chains.size(); ++i) {

646  chains[i].reset(); //remove the chain from our cache.
647  }
648  if (chains[i])
649  active_chains ++;
650  }

662  }

665  conn->recv_data_msg(boost::bind(data_noop_cb, _1, _2), error);
