456  double tau = 0;
457 
458  vector<double> vals;
459  vals.reserve(candidates.size());
460  std::map<DimensionKey, CandidateItem >::iterator iter;
461  for (iter = candidates.begin(); iter != candidates.end(); iter++) {
462  vals.push_back(iter->second.val);
463  }
464  sort (vals.begin(), vals.end());
465 
466  if (vals.size() < num_results)
467  tau = vals[ vals.size() -1];
468  else
469  if ( sort_column[0] == '-' ) //reverse sort; greatest to least
470  tau = vals[vals.size() - num_results];
471  else
472  tau = vals[num_results-1];
473  return tau;
