167 void make_l2_buckets(unsigned int n, std::vector<int> &sequence, unsigned int max=UINT_MAX) {
168  // return 1, 2, ..., 2^31, 3, 6, ..., 3 * 2^30, wrapping around before max
169 
170  sequence.push_back(0);
171  n--;
172 
173  int base = 1;
174  int exp = 0;
175  while (n-- > 0) {
176  sequence.push_back(base * (1 << exp++));
177 
178  if ( unsigned(sequence.back()) > (max >> 1)) {
179  // the next element is always twice the previous element (until wrapping
180  // around), so we wrap around now to avoid overflow.
181 
182  //std::cout << "ending subsequence: " << base << " * 2^" << exp - 1 << std::endl;
183  base += 2; // bases are consecutive odd numbers
184  exp = 0;
185  }
186  }
