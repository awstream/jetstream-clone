240 inline double get_stddev(const vector<long long>& v) {
241  double sum = std::accumulate(v.begin(), v.end(), 0.0);
242  double mean = sum / v.size();
243  std::vector<double> diff(v.size());
244  std::transform(v.begin(), v.end(), diff.begin(),
245  std::bind2nd(std::minus<double>(), mean));
246  double sq_sum = std::inner_product(diff.begin(), diff.end(), diff.begin(), 0.0);
247  return std::sqrt(sq_sum / v.size());
248 }
