212 template<typename AggregateClass>
213 void merge_into_aggregate(jetstream::JSSummary &sum_into, const jetstream::JSSummary &sum_update) {
214  if(contains_aggregate<AggregateClass>(sum_into)) {
215  AggregateClass agg_into(sum_into);
216 
217  if(contains_aggregate<AggregateClass>(sum_update)) {
218  AggregateClass agg_update(sum_update);
219  if (agg_into.size() <= agg_update.size())
220  agg_into.merge_in(agg_update);
221  else {
222  AggregateClass dest(sum_update);
223  dest.merge_in(agg_into);
224  agg_into = dest;
225  }
226  }
227  else {
228  for(int i = 0; i < sum_update.items_size(); ++i) {
229  agg_into.add_item(sum_update.items(i), 1);
230  }
231  }
232 
233  agg_into.serialize_to(sum_into);
234  }
235  else if(contains_aggregate<AggregateClass>(sum_update)) {
236  AggregateClass agg_update(sum_update);
237 
238  for(int i = 0; i < sum_into.items_size(); ++i) {
239  agg_update.add_item(sum_into.items(i), 1);
240  }
241 
242  agg_update.serialize_to(sum_into);
243  sum_into.clear_items();
244  }
245  else
246  {
247  assert(0); //LOG(FATAL) << "One of the summaries should be an aggregate";
248  }
249 }
