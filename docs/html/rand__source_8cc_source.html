131 
132 #ifdef USE_SEQ
133  double incr = (slice_max - slice_min) / rate_per_sec;
134 #else
135  boost::mt19937 gen;

137 #endif
138 
139  time_t now = time(NULL);

141  tuples.reserve(BATCH_SIZE);
142 
143  while (tuples_sent++ < BATCH_SIZE) {
144  boost::shared_ptr<Tuple> t(new Tuple);
145 
146 #ifdef USE_SEQ
147  if ( position_in_slice > rand_data[cur_idx] ) { // note endpoint does NOT trigger increment here. That is deliberate.
148  position_in_slice -= rand_data[cur_idx];
149  cur_idx ++;
150  }
151  if ( cur_idx >= rand_data.size()) {
152  cur_idx = start_idx; //wrap around.

156  position_in_slice += incr;
157 
158 #else

160  double my_acc = accum;
161  int i = start_idx;
162  while ( my_acc + rand_data[i] < d) {
163  my_acc += rand_data[i++];
164  }
165  t->add_e()->set_s_val(rand_labels[i]);
166 #endif
167 
168  t->add_e()->set_t_val(now);

396 RandHistOperator::emit_data() {
397 
398  msec_t start_t = get_msec();
399 
400  unsigned tuples_sent = 0;
401 
402  if ( levels.size() > 1)
403  adapt();
404 
405  msec_t now_msec = get_msec();
406  if(schedule && now_msec > (last_schedule_update + schedule_wait) && tuples_per_sec < schedule_max){
407  last_schedule_update = now_msec;
408  tuples_per_sec += schedule_increment;
409  LOG(INFO) << "Setting tuples per sec " << tuples_per_sec;
410  }
411  unsigned tuples_per_batch = tuples_per_sec * (wait_per_batch/1000);
412  vector< boost::shared_ptr<Tuple> > tuples;
413  while (tuples_sent++ < tuples_per_batch) {
414  while(queue.size()<1)
415  {
416  LOG(INFO) << "Waiting on tuples";
417  js_usleep(100);
418  }
419  {
420  boost::unique_lock<boost::mutex> lock(internals);
421  boost::shared_ptr<Tuple> t = queue.front();
422  queue.pop();
423  tuples.push_back(t);
424  }
425  }
426  msec_t end_t = get_msec();
427  msec_t running_time = (end_t+10) - start_t;
428  if (running_time > wait_per_batch / 2) {
429  LOG_FIRST_N(WARNING, 10) << "Took " << running_time << " to send; exceeds half of wait-per-batch";
430  }
431  if (running_time > wait_per_batch)
432  LOG(FATAL) << "Generation took too long. Runtime "<< running_time << " should be under "<< wait_per_batch;
433 
434  DataplaneMessage end_msg;
435  if (batch ++ % batches_per_window == 0) {
436  end_msg.set_type(DataplaneMessage::END_OF_WINDOW);
437  end_msg.set_window_length_ms(wait_per_batch * batches_per_window-window_fudge_factor);
438  }
439  chain->process(tuples, end_msg);
440 
441  return (wait_per_batch - running_time); //keep running indefinitely
