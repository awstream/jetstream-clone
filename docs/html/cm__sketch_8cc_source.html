195 count_val_t
196 CMMultiSketch::hash_range(unsigned lower, unsigned upper) const {
197 //The model is that we work up the hierarchy, at each time trimming off the ends
198 // and then moving up.
199  count_val_t sum = 0;
200  for (unsigned int level = 0; level < LEVELS + EXACT_LEVELS ; ++level) {
201 // uint32_t true_lower = (lower<< level *BITS_PER_LEVEL);
202 // uint32_t true_upper = (upper<< level *BITS_PER_LEVEL);
203 // cout << "range query [" << true_lower << "," << true_upper<<"] at level "<< level << "\n";
204 
205  int64_t next_level_start = lower & (~(0UL) << BITS_PER_LEVEL );
206  if ( lower != next_level_start) //round up to next power of two
207  next_level_start += (1 << BITS_PER_LEVEL);
208 
209  int64_t next_level_end = (upper & (~(0UL) << BITS_PER_LEVEL )); //this endpoint is NOT included in next level
210  //note that upper & mask can be zero, meaning there's no power of two before this range
211 
212  if (next_level_start < next_level_end) {
213  // cout << " next level will be [" << next_level_start << ", "<< next_level_end <<")\n";
214  assert ( (next_level_start - lower) < (1 << (BITS_PER_LEVEL )));
215 
216  for (int64_t j = lower; j < next_level_start; j++)
217  sum += contrib_from_level(level, j);
218 
219  assert ( (upper - next_level_end) < (1 << (BITS_PER_LEVEL )));
220 
221  for (int64_t j = next_level_end ; j <= upper; ++j)
222  sum += contrib_from_level(level, j);
223  } else {
224 // cout << " last level; [" << lower << "-" << upper << ")\n";
225  assert ( (upper - lower) < (1 << (BITS_PER_LEVEL + 1)));
226  for(int64_t j =lower; j <= upper; ++j)
227  sum += contrib_from_level(level, j);
228  break;
229  }
230 
231  lower = next_level_start >> BITS_PER_LEVEL;
232  upper = (next_level_end-1) >> BITS_PER_LEVEL ;
233 
234  }
235  if (sum > panes[0].total_count)
236  return panes[0].total_count;
237  else
238  return sum;
239 }
