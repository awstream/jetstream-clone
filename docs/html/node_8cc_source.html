616  chain->add_member(nextMember);
617  nextMember->add_chain(chain);
618

628  dataConnMgr.register_new_adaptor(xceiver);

644  newChains.push_back(chain.get());

647  sourcelessChain[ toStart[i] ] = chain;
648  }
