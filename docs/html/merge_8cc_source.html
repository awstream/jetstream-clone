74 size_t fill_buffer(const jetstream::JSSummary &summary, storage *store) {
75  size_t msg_sz = summary.ByteSize();
76 
77  if(store->max_buffer_size < msg_sz) {
78  size_t min_sz = msg_sz < store->max_buffer_size * 2? store->max_buffer_size*2: msg_sz; //max(msg_sz, store->max_buffer_size * 2)
79  char *buffer = (char *) realloc(store->buffer, min_sz*sizeof(char));
80 
81  if(buffer == NULL) {
82  return 0;
83  }
84 
85  store->buffer = buffer;
86  store->max_buffer_size = min_sz;
87  }
88 
89  summary.SerializeToArray(store->buffer, msg_sz);
90  return msg_sz;
91 }

170 char* merge(UDF_INIT *initid, UDF_ARGS *args, char* result, unsigned long* length, char *is_null, char *error) {
171  storage *store= (storage *)initid->ptr;
172  jetstream::JSSummary * sum_ptr = store->summary_ptr;
173 
174  if (NULL == sum_ptr) {
175  *length = 0;
176  *is_null = 1;
177  return NULL;
178  }
179 
180  size_t msg_sz = fill_buffer(*sum_ptr, store);
181 
182  if(0 == msg_sz) {
183  *error = 1;
184  *is_null = 0;
185  *length = 0;
186  return NULL;
187  }
188 
189  *length = msg_sz;
190  *is_null = 0;
191  *error = 0;
192  return store->buffer;
193 }
