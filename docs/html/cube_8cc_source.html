310 DataCube::do_process( OperatorChain * chain,
311  boost::shared_ptr<Tuple> t,
312  DimensionKey key,
313  boost::shared_ptr<std::vector<unsigned int> > levels,
314  boost::shared_ptr<cube::TupleBatch> &tupleBatcher,

317  return;

319 
320  bool in_batch = false;
321 
322  VLOG(2) << "Processing " << key << " thread id " << boost::this_thread::get_id();
323 
324  if (tupleBatcher->contains(key)) {
325  in_batch = true;
326  }
327 
328  boost::shared_ptr<TupleProcessingInfo> tpi;
329  if(!in_batch) {
330  tpi = make_shared<TupleProcessingInfo>(t, key, levels);
331  }
332  else {

334  merge_tuple_into(*(tpi->t), *t);

336

338  { //lock

341  it != subscribers.end(); ++it) {
342  boost::shared_ptr<jetstream::cube::Subscriber> sub = (*it).second;
343  cube::Subscriber::Action act = sub->action_on_tuple(chain, t);
344  if(!in_batch) {
345  tpi->need_new_value = sub->need_new_value(t);
346  tpi->need_old_value = sub->need_old_value(t);
347  }
348  //LOG(INFO) << "Action: "<< act << "send is: " << cube::Subscriber::SEND;
349  if(act == cube::Subscriber::SEND) {
350  if(!in_batch) {
351  VLOG(3) << "Action: "<< act << " adding to insert: " << tpi->key;
352  tpi->insert.push_back((*it).first);
353  }
354  }
355  else if(act == cube::Subscriber::SEND_NO_BATCH) {
356  if(!in_batch) {
357  VLOG(3) << "Action: "<< act << " adding to insert: " << tpi->key;
358  tpi->insert.push_back((*it).first);
359  }
360  can_batch = false;
361  }
362  else if(act == cube::Subscriber::SEND_UPDATE) {
363  if(!in_batch) {
364  VLOG(3) << "Action: "<< act << " adding to update: " << tpi->key;
365  tpi->update.push_back((*it).first);
366  }
367  }
368  }
369  } //lock
370 
371  VLOG(2) <<"Process: "<< key << "in batch: "<<in_batch << " can batch:" << can_batch << " need new:" << tpi->need_new_value << " need old:"<< tpi->need_old_value;
372 
373  //LOG(INFO) << "In do_process. " << elements_in_batch ;
374 
375  bool was_empty=tupleBatcher->is_empty();
376 
377  if(!in_batch) {
378  tupleBatcher->insert_tuple(tpi, can_batch);
379  if(can_batch)
380  flushCongestMon->report_insert(tpi.get(), 1);
381  }
382  else {
383 // LOG_IF(tpi->)
384  tupleBatcher->update_batched_tuple(tpi, can_batch);
385  }
386 
387  if(!tupleBatcher->is_empty() && was_empty)

389  proc->check_flush();
390  }
391
