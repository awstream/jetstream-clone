30  if(inserts != newi)

33  }

38  if ( ((int32_t)inserts ) < -availRoom) // Signed compare: queue won't drain fast enough enough room even with no inserts
39  ratio = 0; //should stop sends altogether to let things drain
40  else

44  ratio = fmin(ratio, max_per_sec / rate_per_sec);
45  removes = inserts - queueDelta;

199 SmoothingQCongestionMonitor::capacity_ratio() {
200  msec_t now = get_msec();
201  msec_t tdelta = now - tstamps[v_idx];
202 
203  if (tdelta >= SMOOTH_STEP_MS) {
204  boost::unique_lock<boost::recursive_mutex> lock(internals);
205 
206  unsigned my_inserts = write_and_clear(&insertsInPeriod);
207  unsigned my_removes = write_and_clear(&removesInPeriod);
208 
209  unsigned newQLen = queueLen + my_inserts - my_removes;
210  atomic_write32(&queueLen, newQLen);
211 
212  v_idx = (v_idx + 1) % WIND_SIZE;
213  inserts[v_idx] = my_inserts;
214  removes[v_idx] = my_removes;
215  tstamps[v_idx] = now;
216 
217  total_inserts = std::accumulate(inserts.begin(),inserts.end(),0);
218  total_removes = std::accumulate(removes.begin(),removes.end(),0);
219  long growth = (total_inserts - total_removes);
220  long future_queue_size = newQLen + growth * PROJECT_STEPS / WIND_SIZE;
221  double delta_to_achieve = queueTarget - future_queue_size; //positive delta means "ramp up";
222  //negative delta means to shrink
223  //inserts per timestep would be total_inserts / WIND_SIZE.
224  //ratio is (delta_to_achieve / PROJECT_STEPS) / inserts_per_timestep
225  if (total_inserts == 0)
226  ratio = INFINITY;
227  else
228  ratio = 1 + ( delta_to_achieve / total_inserts) * (WIND_SIZE / PROJECT_STEPS);
229  if (ratio < 0)
230  ratio = 0;
231 
232  msec_t tdelta_full_window = tstamps[v_idx] - tstamps[(v_idx+1) % WIND_SIZE];
233 
234  double rate_per_sec = total_inserts * 1000.0 / tdelta_full_window;
235  ratio = fmin(ratio, max_per_sec / rate_per_sec);
236  }
237  return fmin(ratio, downstream_status);
238 }
